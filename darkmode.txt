
function darkMode() {
    let dark = document.querySelector("#darkmode")
    let btn_close = document.querySelector(".btn-close")
    let btn_save = document.querySelector(".btn-save")
    let popup = document.querySelector("#pop-up-add")

    let section = document.getElementsByTagName("section")[0]
    let body = document.body

    if (dark.checked == true) {        
        body.classList.add("dark-mode-contrast")
        section.classList.add("dark-mode")
        popup.classList.add("dark-mode")
        btn_save.classList.add("dark-mode-btn")
        btn_close.classList.add("dark-mode-btn")
    } else {
        body.classList.remove("dark-mode-contrast")
        section.classList.remove("dark-mode")
        popup.classList.remove("dark-mode")
        btn_save.classList.remove("dark-mode-btn")
        btn_close.classList.remove("dark-mode-btn")        
    }
    
}
